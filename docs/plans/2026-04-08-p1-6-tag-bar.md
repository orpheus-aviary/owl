# P1-6 Tag Bar Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a Tag Bar component at the bottom of the editor area that allows users to view, add, edit, and delete tags on notes — with autocomplete for `#` hashtags, date-time picker for `/time` and `/alarm`, and frequency modifier dropdown for `/daily` etc.

**Architecture:** Tags are stored per-tab in `useEditorStore` (new `tags` and `originalTags` fields). The Tag Bar renders tag chips with `x` delete buttons + an input field. Input dispatches to type-specific UI: autocomplete popover for `#`, date-time picker popover for `/time`/`/alarm`, and instant add for frequency modifiers. On `Cmd+S`, tags are serialized to raw strings and sent alongside content to `PUT /notes/:id`.

**Tech Stack:** React, zustand, shadcn/ui (Badge, Popover, Calendar, Input, Button), lucide-react icons, Tailwind CSS v4.

```typescript
    // Serialize NoteTag[] to raw tag strings for the API
    const rawTags = tab.tags.map((t) => {
```

```tsx
import type { NoteTag } from '@/lib/api';
import { Badge } from '@/components/ui/badge';
import { X, Clock, Bell, Repeat } from 'lucide-react';

// Display format for different tag types
function formatTagLabel(tag: NoteTag): string {
  switch (tag.tagType) {
    case '#':
      return `#${tag.tagValue}`;
    case '/time': {
      if (!tag.tagValue) return '/time';
      const d = new Date(tag.tagValue);
      return `${(d.getMonth() + 1).toString().padStart(2, '0')}-${d.getDate().toString().padStart(2, '0')} ${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}`;
    }
    case '/alarm': {
      if (!tag.tagValue) return '/alarm';
      const d = new Date(tag.tagValue);
      return `${(d.getMonth() + 1).toString().padStart(2, '0')}-${d.getDate().toString().padStart(2, '0')} ${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}`;
    }
    case '/daily':
      return '每日';
    case '/weekly':
      return '每周';
    case '/monthly':
      return '每月';
    case '/yearly':
      return '每年';
    default:
      return tag.tagValue ?? tag.tagType;
  }
}

function tagIcon(tagType: string) {
  switch (tagType) {
    case '/time':
      return <Clock className="size-3" />;
    case '/alarm':
      return <Bell className="size-3" />;
    case '/daily':
    case '/weekly':
    case '/monthly':
    case '/yearly':
      return <Repeat className="size-3" />;
    default:
      return null;
  }
}

interface TagChipProps {
  tag: NoteTag;
  onDelete: () => void;
  onClick?: () => void;
}

export function TagChip({ tag, onDelete, onClick }: TagChipProps) {
  const isClickable = tag.tagType === '/time' || tag.tagType === '/alarm';
  const icon = tagIcon(tag.tagType);

  return (
    <Badge
      variant="secondary"
      className="gap-1 pl-2 pr-1 py-0.5 text-xs cursor-default select-none"
    >
      {isClickable ? (
        <button
          type="button"
          onClick={onClick}
          className="flex items-center gap-1 hover:underline cursor-pointer"
        >
          {icon}
          <span>{formatTagLabel(tag)}</span>
        </button>
      ) : (
        <span className="flex items-center gap-1">
          {icon}
          <span>{formatTagLabel(tag)}</span>
        </span>
      )}
      <button
        type="button"
        onClick={(e) => {
          e.stopPropagation();
          onDelete();
        }}
        className="ml-0.5 rounded-full p-0.5 hover:bg-muted-foreground/20 transition-colors"
      >
        <X className="size-3" />
      </button>
    </Badge>
  );
}
```

**Step 1: Create the DateTimePicker**

```tsx
import { Calendar } from '@/components/ui/calendar';
import { Input } from '@/components/ui/input';
import { Button } from '@/components/ui/button';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { useState, useEffect } from 'react';

interface DateTimePickerProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  trigger: React.ReactNode;
  initialDate?: Date;
  onConfirm: (date: Date) => void;
}

export function DateTimePicker({ open, onOpenChange, trigger, initialDate, onConfirm }: DateTimePickerProps) {
  const [selectedDate, setSelectedDate] = useState<Date | undefined>(initialDate ?? new Date());
  const [timeValue, setTimeValue] = useState(() => {
    if (initialDate) {
      return `${initialDate.getHours().toString().padStart(2, '0')}:${initialDate.getMinutes().toString().padStart(2, '0')}`;
    }
    return '';
  });

  // Reset when opening with new initialDate
  useEffect(() => {
    if (open) {
      const d = initialDate ?? new Date();
      setSelectedDate(d);
      if (initialDate) {
        setTimeValue(`${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}`);
      } else {
        setTimeValue('');
      }
    }
  }, [open, initialDate]);

  const handleConfirm = () => {
    if (!selectedDate) return;
    const [hours, minutes] = (timeValue || '00:00').split(':').map(Number);
    const result = new Date(selectedDate.getFullYear(), selectedDate.getMonth(), selectedDate.getDate(), hours || 0, minutes || 0, 0, 0);
    onConfirm(result);
    onOpenChange(false);
  };

  return (
    <Popover open={open} onOpenChange={onOpenChange}>
      <PopoverTrigger asChild>{trigger}</PopoverTrigger>
      <PopoverContent side="top" align="start" className="w-auto p-0">
        <div className="p-3 space-y-3">
          <Calendar
            mode="single"
            selected={selectedDate}
            onSelect={setSelectedDate}
          />
          <div className="flex items-center gap-2 px-1">
            <span className="text-sm text-muted-foreground shrink-0">时间</span>
            <Input
              type="time"
              value={timeValue}
              onChange={(e) => setTimeValue(e.target.value)}
              className="h-8 text-sm"
            />
          </div>
          <div className="flex justify-end gap-2 px-1">
            <Button variant="ghost" size="sm" onClick={() => onOpenChange(false)}>
              取消
            </Button>
            <Button size="sm" onClick={handleConfirm}>
              确认
            </Button>
          </div>
        </div>
      </PopoverContent>
    </Popover>
  );
}
```

## Task 4: TagBar component (core)

**Files:**
- Create: `packages/gui/src/renderer/src/components/TagBar.tsx`

This is the main component: chip list + input field + autocomplete + date picker integration.

**Step 1: Create TagBar**

```tsx
import { useState, useRef, useEffect, useCallback } from 'react';
import type { NoteTag, Tag } from '@/lib/api';
import * as api from '@/lib/api';
import { TagChip } from '@/components/TagChip';
import { DateTimePicker } from '@/components/DateTimePicker';
import { Input } from '@/components/ui/input';
import { useTagStore } from '@/stores/tag-store';

const FREQUENCY_TAGS = ['/daily', '/weekly', '/monthly', '/yearly'] as const;
const FREQUENCY_LABELS: Record<string, string> = {
  '/daily': '每日',
  '/weekly': '每周',
  '/monthly': '每月',
  '/yearly': '每年',
};

interface TagBarProps {
  tags: NoteTag[];
  onTagsChange: (tags: NoteTag[]) => void;
}

export function TagBar({ tags, onTagsChange }: TagBarProps) {
  const inputRef = useRef<HTMLInputElement>(null);
  const [inputValue, setInputValue] = useState('');
  const [suggestions, setSuggestions] = useState<Tag[]>([]);
  const [showSuggestions, setShowSuggestions] = useState(false);
  const [selectedSuggestionIndex, setSelectedSuggestionIndex] = useState(0);

  // Date picker state
  const [datePickerOpen, setDatePickerOpen] = useState(false);
  const [datePickerTagType, setDatePickerTagType] = useState<'/time' | '/alarm'>('/time');
  const [editingTagIndex, setEditingTagIndex] = useState<number | null>(null);
  const [editingDate, setEditingDate] = useState<Date | undefined>(undefined);

  // Frequency dropdown
  const [showFrequencyDropdown, setShowFrequencyDropdown] = useState(false);

  const fetchTags = useTagStore((s) => s.fetchTags);

  // Fetch tags on mount for autocomplete
  useEffect(() => {
    fetchTags();
  }, [fetchTags]);

  // Handle input changes — determine which UI to show
  const handleInputChange = useCallback(async (value: string) => {
    setInputValue(value);
    setSelectedSuggestionIndex(0);

    if (value.startsWith('#') && value.length > 1) {
      // Hashtag autocomplete
      try {
        const res = await api.listTags(value.slice(1));
        const existing = new Set(tags.filter((t) => t.tagType === '#').map((t) => t.tagValue));
        setSuggestions((res.data ?? []).filter((t) => !existing.has(t.tagValue)));
        setShowSuggestions(true);
      } catch {
        setSuggestions([]);
      }
      setShowFrequencyDropdown(false);
    } else if (value === '/') {
      // Show frequency + time options
      setShowFrequencyDropdown(true);
      setShowSuggestions(false);
    } else if (value.startsWith('/time') || value.startsWith('/alarm')) {
      // Time-based tag — will open date picker on Enter
      setShowSuggestions(false);
      setShowFrequencyDropdown(false);
    } else {
      setShowSuggestions(false);
      setShowFrequencyDropdown(false);
    }
  }, [tags]);

  // Add a tag (dedup by tagType:tagValue)
  const addTag = useCallback((tagType: string, tagValue: string | null) => {
    const key = `${tagType}:${tagValue ?? ''}`;
    if (tags.some((t) => `${t.tagType}:${t.tagValue ?? ''}` === key)) return;
    // Use a temp id — server will assign real id on save
    const newTag: NoteTag = { id: `temp-${Date.now()}`, tagType, tagValue };
    onTagsChange([...tags, newTag]);
    setInputValue('');
    setShowSuggestions(false);
    setShowFrequencyDropdown(false);
  }, [tags, onTagsChange]);

  // Delete a tag by index
  const deleteTag = useCallback((index: number) => {
    onTagsChange(tags.filter((_, i) => i !== index));
  }, [tags, onTagsChange]);

  // Handle Enter key
  const handleKeyDown = useCallback((e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Escape') {
      setShowSuggestions(false);
      setShowFrequencyDropdown(false);
      setInputValue('');
      return;
    }

    if (e.key === 'ArrowDown') {
      e.preventDefault();
      if (showSuggestions) {
        setSelectedSuggestionIndex((prev) => Math.min(prev + 1, suggestions.length - 1));
      }
      return;
    }

    if (e.key === 'ArrowUp') {
      e.preventDefault();
      if (showSuggestions) {
        setSelectedSuggestionIndex((prev) => Math.max(prev - 1, 0));
      }
      return;
    }

    if (e.key === 'Enter') {
      e.preventDefault();

      // If suggestion is selected, use it
      if (showSuggestions && suggestions.length > 0) {
        const selected = suggestions[selectedSuggestionIndex];
        if (selected) {
          addTag('#', selected.tagValue);
          return;
        }
      }

      const trimmed = inputValue.trim();
      if (!trimmed) return;

      // Hashtag
      if (trimmed.startsWith('#') && trimmed.length > 1) {
        addTag('#', trimmed.slice(1));
        return;
      }

      // Frequency modifier
      if (FREQUENCY_TAGS.includes(trimmed.toLowerCase() as typeof FREQUENCY_TAGS[number])) {
        addTag(trimmed.toLowerCase(), '');
        return;
      }

      // Time-based → open date picker
      if (trimmed.startsWith('/time') || trimmed.startsWith('/alarm')) {
        const isAlarm = trimmed.startsWith('/alarm');
        setDatePickerTagType(isAlarm ? '/alarm' : '/time');
        setEditingTagIndex(null);
        // Try to parse any inline date
        const dateStr = trimmed.slice(isAlarm ? 6 : 5).trim();
        if (dateStr) {
          const parts = dateStr.split(/[\sT]+/);
          const dateParts = parts[0]?.split('-').map(Number);
          if (dateParts && dateParts.length >= 2) {
            const now = new Date();
            const year = dateParts.length === 3 ? (dateParts[0] < 100 ? 2000 + dateParts[0] : dateParts[0]) : now.getFullYear();
            const month = dateParts.length === 3 ? dateParts[1] : dateParts[0];
            const day = dateParts.length === 3 ? dateParts[2] : dateParts[1];
            const timeParts = parts[1]?.split(':').map(Number);
            setEditingDate(new Date(year, month - 1, day, timeParts?.[0] ?? 0, timeParts?.[1] ?? 0));
          } else {
            setEditingDate(undefined);
          }
        } else {
          setEditingDate(undefined);
        }
        setDatePickerOpen(true);
        return;
      }
    }
  }, [inputValue, showSuggestions, suggestions, selectedSuggestionIndex, addTag]);

  // Handle frequency option click
  const handleFrequencyClick = useCallback((freq: string) => {
    if (freq === '/time' || freq === '/alarm') {
      setDatePickerTagType(freq as '/time' | '/alarm');
      setEditingTagIndex(null);
      setEditingDate(undefined);
      setDatePickerOpen(true);
      setShowFrequencyDropdown(false);
      setInputValue('');
    } else {
      addTag(freq, '');
    }
  }, [addTag]);

  // Handle date picker confirm (new tag)
  const handleDateConfirm = useCallback((date: Date) => {
    const pad = (n: number, w = 2) => n.toString().padStart(w, '0');
    const isoValue = `${pad(date.getFullYear(), 4)}-${pad(date.getMonth() + 1)}-${pad(date.getDate())}T${pad(date.getHours())}:${pad(date.getMinutes())}:00`;

    if (editingTagIndex !== null) {
      // Editing existing time tag
      const updated = tags.map((t, i) =>
        i === editingTagIndex ? { ...t, tagValue: isoValue } : t,
      );
      onTagsChange(updated);
    } else {
      addTag(datePickerTagType, isoValue);
    }
    setEditingTagIndex(null);
    setInputValue('');
  }, [tags, onTagsChange, addTag, datePickerTagType, editingTagIndex]);

  // Handle click on time chip → open date picker for editing
  const handleTimeChipClick = useCallback((index: number) => {
    const tag = tags[index];
    if (!tag || (tag.tagType !== '/time' && tag.tagType !== '/alarm')) return;
    setDatePickerTagType(tag.tagType as '/time' | '/alarm');
    setEditingTagIndex(index);
    if (tag.tagValue) {
      setEditingDate(new Date(tag.tagValue));
    } else {
      setEditingDate(undefined);
    }
    setDatePickerOpen(true);
  }, [tags]);

  // Expose focus method via ref for Cmd+L
  // We'll use a global event approach instead (see Task 5)

  return (
    <div className="flex items-center gap-1.5 flex-wrap px-3 py-2 border-t border-border min-h-[40px] shrink-0">
      {/* Tag chips */}
      {tags.map((tag, index) => (
        <TagChip
          key={`${tag.tagType}:${tag.tagValue}:${index}`}
          tag={tag}
          onDelete={() => deleteTag(index)}
          onClick={
            tag.tagType === '/time' || tag.tagType === '/alarm'
              ? () => handleTimeChipClick(index)
              : undefined
          }
        />
      ))}

      {/* Input + autocomplete + frequency dropdown + date picker */}
      <div className="relative flex-1 min-w-[120px]">
        <DateTimePicker
          open={datePickerOpen}
          onOpenChange={setDatePickerOpen}
          initialDate={editingDate}
          onConfirm={handleDateConfirm}
          trigger={
            <Input
              ref={inputRef}
              data-tag-input
              value={inputValue}
              onChange={(e) => handleInputChange(e.target.value)}
              onKeyDown={handleKeyDown}
              onFocus={() => {
                if (inputValue.startsWith('#') && inputValue.length > 1) {
                  setShowSuggestions(true);
                }
              }}
              onBlur={() => {
                // Delay to allow click on suggestions
                setTimeout(() => {
                  setShowSuggestions(false);
                  setShowFrequencyDropdown(false);
                }, 150);
              }}
              placeholder="输入标签..."
              className="h-7 text-xs border-none shadow-none bg-transparent px-1 focus-visible:ring-0"
            />
          }
        />

        {/* Hashtag autocomplete dropdown */}
        {showSuggestions && suggestions.length > 0 && (
          <div className="absolute bottom-full left-0 mb-1 w-48 bg-popover border border-border rounded-md shadow-md z-50 max-h-40 overflow-y-auto">
            {suggestions.map((tag, i) => (
              <button
                key={tag.id}
                type="button"
                className={`w-full text-left px-3 py-1.5 text-xs hover:bg-accent transition-colors ${i === selectedSuggestionIndex ? 'bg-accent' : ''}`}
                onMouseDown={(e) => {
                  e.preventDefault(); // prevent blur
                  addTag('#', tag.tagValue);
                }}
              >
                #{tag.tagValue}
              </button>
            ))}
          </div>
        )}

        {/* Frequency / time type dropdown */}
        {showFrequencyDropdown && (
          <div className="absolute bottom-full left-0 mb-1 w-48 bg-popover border border-border rounded-md shadow-md z-50">
            {['/time', '/alarm', ...FREQUENCY_TAGS].map((opt) => (
              <button
                key={opt}
                type="button"
                className="w-full text-left px-3 py-1.5 text-xs hover:bg-accent transition-colors"
                onMouseDown={(e) => {
                  e.preventDefault();
                  handleFrequencyClick(opt);
                }}
              >
                {opt === '/time' ? '/time (过期时间)' : opt === '/alarm' ? '/alarm (提醒)' : FREQUENCY_LABELS[opt] ?? opt}
              </button>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
```
