# P2-5c Folder / Note Drag-and-Drop Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add dnd-kit drag-and-drop to the folder sidebar panel — reorder folders, move folders across parents, drop notes into folders, and clear a note's folder by dropping into blank space.

**Architecture:** `@dnd-kit/core` with a single top-level `DndContext` in `App.tsx`, a custom `DragOverlay` for VSCode-style drag visuals, and three vertically-stacked droppable zones per folder row (top/middle/bottom) to distinguish "insert above", "drop into", and "insert below". Reuses existing daemon endpoints (`PATCH /folders/reorder` already accepts per-item `parent_id`, so cross-parent moves are one atomic call).

**Tech Stack:** React + zustand + @dnd-kit/core (new dep) + existing folder/editor stores.

**Design doc:** `docs/plans/2026-04-15-p2-5c-dnd-design.md`

```ts
/**
 * Returns true iff `targetId` lives anywhere in the subtree rooted at
 * `ancestorId`. Used to reject drags that would create a cycle
 * (dragging a folder into its own descendant).
 */
export function isDescendant(
  folders: Folder[],
  ancestorId: string,
  targetId: string,
): boolean {
  if (ancestorId === targetId) return false;
  const childrenByParent = new Map<string, string[]>();
  for (const f of folders) {
    if (!f.parent_id) continue;
    const arr = childrenByParent.get(f.parent_id) ?? [];
    arr.push(f.id);
    childrenByParent.set(f.parent_id, arr);
  }
  const stack = [...(childrenByParent.get(ancestorId) ?? [])];
  while (stack.length) {
    const id = stack.pop();
    if (id === targetId) return true;
    const kids = childrenByParent.get(id!);
    if (kids) stack.push(...kids);
  }
  return false;
}
```

## Task 4: Build drag data contract module

```ts
/** Drag-and-drop data contracts for the folder panel. See docs/plans/2026-04-15-p2-5c-dnd-design.md §3. */

export type DragData =
  | { kind: 'folder'; folderId: string; parentId: string | null }
  | { kind: 'note'; noteId: string };

export type DropTarget =
  | { kind: 'folder-node'; folderId: string }
  | { kind: 'folder-gap'; parentId: string | null; index: number }
  | { kind: 'root-blank' };

/** Type guards for narrowing dnd-kit's `active.data.current` / `over.data.current`. */
export function isDragData(x: unknown): x is DragData {
  return (
    typeof x === 'object' &&
    x !== null &&
    'kind' in x &&
    (x.kind === 'folder' || x.kind === 'note')
  );
}

export function isDropTarget(x: unknown): x is DropTarget {
  return (
    typeof x === 'object' &&
    x !== null &&
    'kind' in x &&
    (x.kind === 'folder-node' || x.kind === 'folder-gap' || x.kind === 'root-blank')
  );
}
```

**Step 2:** Inside `App()`, add:

```tsx
const [activeDrag, setActiveDrag] = useState<DragData | null>(null);
const sensors = useSensors(
  useSensor(MouseSensor, { activationConstraint: { distance: 5 } }),
);

const onDragStart = (e: DragStartEvent) => {
  const data = e.active.data.current;
  if (isDragData(data)) setActiveDrag(data);
};

const onDragEnd = async (e: DragEndEvent) => {
  setActiveDrag(null);
  const drag = e.active.data.current;
  const drop = e.over?.data.current;
  if (!isDragData(drag) || !drop || !isDropTarget(drop)) return;

  const folderStore = useFolderStore.getState();
  const editorStore = useEditorStore.getState();

  // ── Folder drag ───────────────────────────────
  if (drag.kind === 'folder') {
    const { folderId, parentId: srcParent } = drag;

    // Drop on a folder node → become child of that node
    if (drop.kind === 'folder-node') {
      if (drop.folderId === folderId) return; // self
      if (isDescendant(folderStore.folders, folderId, drop.folderId)) return; // cycle
      if (srcParent === drop.folderId) return; // no-op (already a child)
      await folderStore.move(folderId, drop.folderId);
      return;
    }

    // Drop on a gap → sibling reorder (may cross parents)
    if (drop.kind === 'folder-gap') {
      if (drop.parentId === folderId) return;
      if (drop.parentId && isDescendant(folderStore.folders, folderId, drop.parentId)) return;
      // Build ordered sibling list at target parent
      const siblings = folderStore.folders
        .filter((f) => f.parent_id === drop.parentId && f.id !== folderId)
        .sort((a, b) => a.position - b.position);
      siblings.splice(drop.index, 0, { ...folderStore.folders.find((f) => f.id === folderId)!, parent_id: drop.parentId });
      const items = siblings.map((f, i) => ({
        id: f.id,
        parent_id: drop.parentId,
        position: i,
      }));
      await folderStore.reorder(items);
      return;
    }

    // Drop on root blank → promote to top-level (only if not already root)
    if (drop.kind === 'root-blank') {
      if (srcParent === null) return;
      await folderStore.move(folderId, null);
      return;
    }
  }

  // ── Note drag ─────────────────────────────────
  if (drag.kind === 'note') {
    const { noteId } = drag;
    let targetFolderId: string | null | undefined;
    if (drop.kind === 'folder-node') targetFolderId = drop.folderId;
    else if (drop.kind === 'root-blank') targetFolderId = null;
    else return; // gaps are not valid drop targets for notes
    try {
      await (await import('@/lib/api')).moveNoteToFolder(noteId, targetFolderId);
      editorStore.syncTabFolderId(noteId, targetFolderId);
      // Refresh note list so folder filters update
      useNoteStore.getState().fetchNotes();
    } catch (err) {
      console.error('note move failed', err);
    }
  }
};
```
